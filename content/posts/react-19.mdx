### ðŸ§  1. Actions & Enhanced Data Handling

React 19 introduces **Actions** â€” a new way to handle asynchronous mutations and form submissions. Actions automatically manage pending states, optimistic updates, and errors with less boilerplate.

**Example: Handling a form submission with Actions**

```tsx
"use client";
import { useActionState, action } from "react";

const loginAction = action(
  async ({ email, password }: { email: string; password: string }) => {
    const res = await fetch("/api/login", {
      method: "POST",
      body: JSON.stringify({ email, password }),
      headers: { "Content-Type": "application/json" },
    });
    if (!res.ok) throw new Error("Login failed");
    return res.json();
  },
);

export default function LoginForm() {
  const state = useActionState(loginAction);

  return (
    <form
      onSubmit={(e) => {
        e.preventDefault();
        const form = new FormData(e.currentTarget);
        loginAction({
          email: form.get("email") as string,
          password: form.get("password") as string,
        });
      }}
    >
      <input name="email" type="email" placeholder="Email" required />
      <input name="password" type="password" placeholder="Password" required />
      <button type="submit" disabled={state.pending}>
        {state.pending ? "Logging in..." : "Login"}
      </button>
      {state.error && <p style={{ color: "red" }}>{state.error.message}</p>}
    </form>
  );
}
```
